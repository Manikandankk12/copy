pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build Docker Images') {
            steps {
                sh 'docker-compose build'
            }
        }

        stage('Run Containers') {
            steps {
                sh 'docker-compose up -d'
            }
        }

        stage('Test') {
            steps {
                // Example: curl the Flask app to check response
                sh 'curl -f http://localhost:5001 || exit 1'
            }
        }
    }

    post {
        always {
            sh 'docker-compose down'
        }
    }
}
